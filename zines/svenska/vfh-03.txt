define('PROXY_HOST', '127.0.0.1');
define('PROXY_PORT', '8118'); // privoxy,tor

function pr0xy($url){
                $proxy = PROXY_HOST . ':' . PROXY_PORT;
                $hcurl = curl_init(); 
                curl_setopt($hcurl, CURLOPT_PROXY, $proxy);
                curl_setopt($hcurl, CURLOPT_HEADER, 1);  
                curl_setopt($hcurl, CURLOPT_USERAGENT, 
				'Opera/9.2 (Linux 2.6; en-US;)'); 
                curl_setopt($hcurl, CURLOPT_URL, $url);
                curl_setopt($hcurl, CURLOPT_RETURNTRANSFER, 1); 
                curl_setopt($hcurl, CURLOPT_FOLLOWLOCATION, 1);
                curl_setopt($hcurl, CURLOPT_TIMEOUT, '300');
                $ret = curl_exec($hcurl);
                curl_close($hcurl);

                return $ret;
}
